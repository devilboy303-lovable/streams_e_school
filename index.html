// ============================================================ // 스트림즈 (STREAMS) - Google Apps Script Backend // ============================================================ // 설정: 아래 SHEET_ID를 본인 Google Sheet ID로 교체하세요 // Google Sheet URL: https://docs.google.com/spreadsheets/d/[이 부분]/edit // ============================================================ const SHEET_ID = '1yEjeUmIZmtsPPyzk99DOROutPASzpcShq-5cSeFyAMc'; // ============================================================ // HTTP 진입점 // ============================================================ function doGet(e) { const action = e.parameter.action; // action 없으면 웹앱 HTML 서빙 if (!action) { return HtmlService.createHtmlOutputFromFile('index') .setTitle('스트림즈') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } return handleAction(action, e.parameter); } function doPost(e) { const params = JSON.parse(e.postData.contents); return handleAction(params.action, params); } function handleAction(action, params) { let result; try { switch (action) { case 'newGame': result = newGame(params); break; case 'nextNumber': result = nextNumber(params); break; case 'getGame': result = getGame(params); break; case 'endGame': result = endGame(params); break; case 'getHistory': result = getHistory(params); break; case 'deleteGame': result = deleteGame(params); break; default: result = { error: 'unknown action' }; } } catch (err) { result = { error: err.message }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ============================================================ // 게임 로직 // ============================================================ /** * 새 게임 시작 * params: { rounds } - 라운드 수 (기본 5) */ function newGame(params) { const ss = SpreadsheetApp.openById(SHEET_ID); const rounds = parseInt(params.rounds) || 5; const gameId = Utilities.getUuid(); const createdAt = new Date().toISOString(); // 각 라운드별 숫자 시퀀스 미리 생성 (1~24개 랜덤) const allRounds = []; for (let r = 0; r < rounds; r++) { allRounds.push(generateRoundNumbers()); } const gameData = { gameId, createdAt, rounds, currentRound: 1, currentIndex: 0, // 현재 라운드에서 몇 번째 숫자까지 공개했나 status: 'playing', // playing | finished allRounds, // [[숫자배열], [숫자배열], ...] }; // 시트에 저장 const sheet = getOrCreateSheet(ss, 'Games'); sheet.appendRow([ gameId, createdAt, rounds, JSON.stringify(gameData), 'playing' ]); return { ok: true, gameId, gameData }; } /** * 다음 숫자 공개 * params: { gameId } */ function nextNumber(params) { const { gameId } = params; const { sheet, rowIndex, gameData } = loadGame(gameId); if (gameData.status === 'finished') { return { ok: false, error: '게임이 종료되었습니다' }; } const round = gameData.currentRound - 1; // 0-indexed const numbers = gameData.allRounds[round]; if (gameData.currentIndex >= numbers.length) { // 이 라운드 숫자 다 공개됨 return { ok: false, error: '이미 모든 숫자가 공개되었습니다', roundDone: true }; } gameData.currentIndex++; saveGame(sheet, rowIndex, gameData); return { ok: true, revealed: numbers.slice(0, gameData.currentIndex), current: numbers[gameData.currentIndex - 1], total: numbers.length, index: gameData.currentIndex, roundDone: gameData.currentIndex >= numbers.length }; } /** * 게임 상태 조회 * params: { gameId } */ function getGame(params) { const { gameId } = params; const { gameData } = loadGame(gameId); const round = gameData.currentRound - 1; const numbers = gameData.allRounds[round]; return { ok: true, gameData, revealed: numbers.slice(0, gameData.currentIndex), total: numbers.length, roundDone: gameData.currentIndex >= numbers.length }; } /** * 라운드 종료 / 다음 라운드로 * params: { gameId } */ function endGame(params) { const { gameId, action: subAction } = params; const { sheet, rowIndex, gameData } = loadGame(gameId); if (subAction === 'nextRound') { if (gameData.currentRound >= gameData.rounds) { gameData.status = 'finished'; } else { gameData.currentRound++; gameData.currentIndex = 0; } } else { gameData.status = 'finished'; } saveGame(sheet, rowIndex, gameData); // 게임 종료면 히스토리 시트에도 기록 if (gameData.status === 'finished') { recordHistory(gameData); } return { ok: true, gameData }; } /** * 날짜별 게임 기록 조회 * params: { date } - 'YYYY-MM-DD' (없으면 전체) */ function getHistory(params) { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = getOrCreateSheet(ss, 'History'); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return { ok: true, history: [] }; const rows = data.slice(1); // 헤더 제외 let history = rows.map(r => ({ date: r[0], gameId: r[1], rounds: r[2], completedAt: r[3], summary: r[4] ? JSON.parse(r[4]) : [] })); if (params.date) { history = history.filter(h => h.date === params.date); } return { ok: true, history: history.reverse() }; } /** * 게임 기록 삭제 */ function deleteGame(params) { const { gameId } = params; const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = getOrCreateSheet(ss, 'Games'); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === gameId) { sheet.deleteRow(i + 1); return { ok: true }; } } return { ok: false, error: '게임을 찾을 수 없습니다' }; } // ============================================================ // 내부 유틸 // ============================================================ function generateRoundNumbers() { // 카드 풀: 1~10 각 1장, 11~19 각 2장, 20~30 각 1장, ★ 조커 1장 = 총 40장 // 라운드당 20장 랜덤 뽑기 const pool = []; for (let i = 1; i <= 10; i++) pool.push(String(i)); for (let i = 11; i <= 19; i++) { pool.push(String(i)); pool.push(String(i)); } for (let i = 20; i <= 30; i++) pool.push(String(i)); pool.push('★'); // Fisher-Yates shuffle for (let i = pool.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [pool[i], pool[j]] = [pool[j], pool[i]]; } return pool.slice(0, 20); // 20장 공개 } function loadGame(gameId) { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = getOrCreateSheet(ss, 'Games'); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === gameId) { return { sheet, rowIndex: i + 1, gameData: JSON.parse(data[i][3]) }; } } throw new Error('게임을 찾을 수 없습니다: ' + gameId); } function saveGame(sheet, rowIndex, gameData) { sheet.getRange(rowIndex, 4).setValue(JSON.stringify(gameData)); sheet.getRange(rowIndex, 5).setValue(gameData.status); } function recordHistory(gameData) { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = getOrCreateSheet(ss, 'History'); // 헤더 없으면 추가 if (sheet.getLastRow() === 0) { sheet.appendRow(['날짜', '게임ID', '라운드수', '완료시각', '요약']); } const today = Utilities.formatDate(new Date(), 'Asia/Seoul', 'yyyy-MM-dd'); const now = new Date().toISOString(); // 라운드별 숫자 요약 const summary = gameData.allRounds.map((nums, i) => ({ round: i + 1, numbers: nums })); sheet.appendRow([today, gameData.gameId, gameData.rounds, now, JSON.stringify(summary)]); } function getOrCreateSheet(ss, name) { let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); if (name === 'Games') { sheet.appendRow(['gameId', 'createdAt', 'rounds', 'data', 'status']); } else if (name === 'History') { sheet.appendRow(['날짜', '게임ID', '라운드수', '완료시각', '요약']); } } return sheet; }
